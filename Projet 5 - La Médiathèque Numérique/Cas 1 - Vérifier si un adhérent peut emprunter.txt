FONCTION verifEmprunt(ret, med)
   SI ret >= 1
      AFFICHER "Nouvel emprunt interdit, car" + ret + "retard en cours"
   FIN SI
   SINON SI med >= 5
      AFFICHER "Nouvel emprunt interdit, déjà" + med + "emprunts en cours"
   FIN SI
   SINON
      compteur <- 5 - med
      AFFICHER "Le client a le droit d'emprunter" + compteur + "medias"
   FIN SI
FIN FONCTION

ALGORITHME PRINCIPAL
DEMANDER à la BDD les données retard et media du client
VerificationEmprunt <- verifEmprunt(retard, media)
   AFFICHER VerificationEmprunt
